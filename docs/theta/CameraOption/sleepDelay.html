Future<Map<String, dynamic>> sleepDelay (
  1. int seconds

auto sleep. requires a number between 60 and 65535 the default value is 300 seconds or 5 minutes A value of 65535 will disable sleep

Implementation

static Future<Map<String, dynamic>> sleepDelay(int seconds) async {
  var data = {
    'name': 'camera.setOptions',
    'parameters': {
      'options': {'sleepDelay': seconds}
    }
  };
  var response = connect(_url, 'post', data);
  return response;
}
